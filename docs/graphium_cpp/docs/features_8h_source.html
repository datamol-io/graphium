151template<> struct FeatureValues<int16_t> {
152 static constexpr int16_t zero = 0x0000;
153 static constexpr int16_t one = 0x3C00;
154 static constexpr int16_t nan_value = 0x7C01;
155
156 template<typename T>
157 static int16_t convertToFeatureType(T inputType) {
158 static_assert(std::is_floating_point_v<T>);
159 return c10::detail::fp16_ieee_from_fp32_value(float(inputType));
160 }
161
162 static constexpr bool is_finite(int16_t v) {
163 // If the exponent bits are the maximum value, v is infinite or NaN
164 return (v & 0x7C00) != 0x7C00;
165 }
166
167 using MathType = float;
168};

208template<typename T>
209constexpr int64_t mask_nans(T* data, size_t n, MaskNaNStyle style, T value) {
210 if (style == MaskNaNStyle::NONE) {
211 return 0;
212 }
213 if (style == MaskNaNStyle::REPLACE) {
214 for (size_t i = 0; i < n; ++i) {
215 if (!FeatureValues<T>::is_finite(data[i])) {
216 data[i] = value;
217 }
218 }
219 return 0;
220 }
221
222 assert(style == MaskNaNStyle::REPORT);
223 int64_t num_nans = 0;
224 for (size_t i = 0; i < n; ++i) {
225 num_nans += (!FeatureValues<T>::is_finite(data[i]));
226 }
227 return num_nans;
228}
